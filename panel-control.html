function activarNotificaciones() { const q = query(collection(db, "proyectos"), where("notificar_admin", "==", true)); onSnapshot(q, (snap) => { const list = document.getElementById('notifList'); const badge = document.getElementById('notifCount'); const section = document.getElementById('notifSection'); list.innerHTML = ""; const count = snap.size; if (count > 0) { badge.innerText = count; badge.style.display = "block"; section.style.display = "block"; snap.forEach(d => { const p = d.data(); // LÓGICA DE NOMBRES INTELIGENTE let nombreSolicitante = p.solicitante_notificacion || "Un colaborador"; let textoAccion = ""; if(p.solicitud_eliminacion) textoAccion = `${nombreSolicitante} solicita DESASIGNARSE de esta tarea`; else if(p.solicitud_desmarcar) textoAccion = `${nombreSolicitante} solicita REABRIR esta tarea`; else textoAccion = `${nombreSolicitante} reportó un AVANCE en la bitácora`; list.innerHTML += `
${textoAccion.toUpperCase()} ${p.producto} | Cliente: ${p.cliente}
`; }); } else { badge.style.display = "none"; section.style.display = "none"; } }); }
